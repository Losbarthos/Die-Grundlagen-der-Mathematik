\chapter{Einleitung}

Band~03 konstruiert die natürlichen Zahlen innerhalb der Mengenlehre:
\(0\) wird dort durch die leere Menge realisiert, und der Nachfolger entsteht
als \(x\cup\{x\}\). Dieser Band wählt einen anderen internen Ausgangspunkt.
Wir betrachten eine abstrakte Peano-Struktur mit den Zeichen
\[
  \mathbb{N},\qquad 0,\qquad n+1.
\]
Das Zeichen \(n+1\) ist zunächst ein primitives Nachfolgerzeichen. Es ist
noch keine Addition mit der Zahl \(1\). Band~03 dient in diesem Band nur als
metamathematische Rechtfertigung dafür, dass eine solche Struktur modelliert
werden kann. Die Beweise und Definitionen dieses Bandes arbeiten intern mit
den Peano-Axiomen.

\chapter{Peano-Axiome}

Die folgenden Axiome legen die Struktur \((\mathbb{N},0,n\mapsto n+1)\)
fest. Das Nachfolgerzeichen wird im ganzen Abschnitt als primitives
einstelliges Operationszeichen gelesen.

\FormulaAxiomAuto[Null ist natürlich]{0\in\mathbb{N}}

\FormulaAxiomAuto[Nachfolgerabschluss]{n\in\mathbb{N}\vdash n+1\in\mathbb{N}}

\FormulaAxiomAuto[Kein Nachfolger ist Null]{n\in\mathbb{N}\vdash n+1\neq 0}

\FormulaAxiomAuto[Injektivität des Nachfolgers]{n\in\mathbb{N}\dsep m\in\mathbb{N}\dsep n+1=m+1\vdash n=m}

\FormulaAxiomAuto[Induktionsaxiom]{P(0)\dsep \forall n\in\mathbb{N}\bigl(P(n)\rightarrow P(n+1)\bigr)\vdash \forall n\in\mathbb{N}\,P(n)}

\begin{remark}
Die Existenz und Eindeutigkeit eines Nachfolgers ist in dieser Schreibweise
bereits durch die Termbildung \(n+1\) und den Nachfolgerabschluss enthalten.
Wir formulieren sie deshalb nicht zusätzlich als eigenes Axiom.
\end{remark}

\chapter{Rekursion}

\begin{definition}[Rekursiv geschlossene Relationen]
Seien \(A\) eine Menge, \(a_0\in A\) und \(s\colon A\to A\). Eine Relation
\(R\subseteq\mathbb{N}\times A\) heißt zu \((a_0,s)\) rekursiv geschlossen,
wenn
\[
  (0,a_0)\in R
  \quad\text{und}\quad
  (n,a)\in R\vdash (n+1,s(a))\in R
\]
für alle \(n\in\mathbb{N}\) und \(a\in A\) gelten.
\end{definition}

\FormulaDefAuto[Rekursionskern]{\mathcal{R}_{a_0,s}\coloneqq \bigcap\{R\subseteq\mathbb{N}\times A\mid R\text{ ist zu }(a_0,s)\text{ rekursiv geschlossen}\}}

\FormulaThmAuto[Dedekindscher Rekursionssatz]{a_0\in A\dsep s\colon A\to A\vdash \exists! f\colon\mathbb{N}\to A\,\Bigl(f(0)=a_0\land \forall n\in\mathbb{N}\bigl(f(n+1)=s(f(n))\bigr)\Bigr)}
\begin{tabproof}
  \proofstep{1}{a_0\in A}{\rA}
  \proofstep{2}{s\colon A\to A}{\rA}
  \proofstep{1,2}{\mathcal{R}_{a_0,s}\subseteq\mathbb{N}\times A}{\FormulaRefAuto{\mathcal{R}_{a_0,s}\coloneqq \bigcap\{R\subseteq\mathbb{N}\times A\mid R\text{ ist zu }(a_0,s)\text{ rekursiv geschlossen}\}}}
  \proofstep{1,2}{(0,a_0)\in\mathcal{R}_{a_0,s}}{\text{Minimaler rekursiver Abschluss}}
  \proofstep{1,2}{(n,a)\in\mathcal{R}_{a_0,s}\vdash (n+1,s(a))\in\mathcal{R}_{a_0,s}}{\text{Rekursive Geschlossenheit}}
  \proofstep{1,2}{\forall n\in\mathbb{N}\,\exists! a\in A\,((n,a)\in\mathcal{R}_{a_0,s})}{\text{Induktion über \(n\)}}
  \proofstep{1,2}{\exists! f\colon\mathbb{N}\to A\,\Bigl(f(0)=a_0\land \forall n\in\mathbb{N}\bigl(f(n+1)=s(f(n))\bigr)\Bigr)}{\text{Graph von \(\mathcal{R}_{a_0,s}\)}}
\end{tabproof}

\FormulaThmAuto[Parametrisierte Rekursion]{B_0\colon B\to A\dsep H\colon B\times A\to A\vdash \exists! F\colon B\times\mathbb{N}\to A\,\Bigl(F(p,0)=B_0(p)\land F(p,n+1)=H(p,F(p,n))\Bigr)}
\begin{tabproof}
  \proofstep{1}{B_0\colon B\to A}{\rA}
  \proofstep{2}{H\colon B\times A\to A}{\rA}
  \proofstep{1,2}{\exists! f_p\colon\mathbb{N}\to A\,\bigl(f_p(0)=B_0(p)\land f_p(n+1)=H(p,f_p(n))\bigr)}{\FormulaRefAuto{a_0\in A\dsep s\colon A\to A\vdash \exists! f\colon\mathbb{N}\to A\,\Bigl(f(0)=a_0\land \forall n\in\mathbb{N}\bigl(f(n+1)=s(f(n))\bigr)\Bigr)}}
  \proofstep{1,2}{\exists! F\colon B\times\mathbb{N}\to A\,\Bigl(F(p,0)=B_0(p)\land F(p,n+1)=H(p,F(p,n))\Bigr)}{\text{Punktweise Zusammenfassung der \(f_p\)}}
\end{tabproof}

\chapter{Addition}

\section{Definition und Grundgleichungen}

Die Addition wird im zweiten Argument rekursiv definiert. In der Gleichung
\(m+(n+1)=(m+n)+1\) bezeichnet das rechte äußere \(+1\) weiterhin den
primitiven Nachfolger.

\FormulaDefAuto[Addition]{m\in\mathbb{N}\dsep n\in\mathbb{N}\vdash \bigl(m+0=m\land m+(n+1)=(m+n)+1\bigr)}

\FormulaDefAuto[Die Eins]{\mathbf{1}\coloneqq 0+1}

Nach dieser Definition schreiben wir für \(\mathbf{1}\) auch schlicht \(1\).
Die Kompatibilität der beiden Lesarten wird unten bewiesen.

\[
  m\in\mathbb{N}\dsep n\in\mathbb{N}\vdash m+n\in\mathbb{N}.

\chapter{Dezimalsystem}

\section{Ziffern}

Die Zahlzeichen \(0,1,\ldots,10\) werden nun als iterierte Nachfolger
verstanden. Nach der Kompatibilitätsregel \(n+\mathbf{1}=n+1\) dürfen wir
\(+1\) zugleich als Addition mit \(1\) lesen.

\FormulaDefAuto[Dezimale Grundziffern]{\begin{gathered}
1\coloneqq 0+1,\quad
2\coloneqq 1+1,\quad
3\coloneqq 2+1,\quad
4\coloneqq 3+1,\quad
5\coloneqq 4+1,\\
6\coloneqq 5+1,\quad
7\coloneqq 6+1,\quad
8\coloneqq 7+1,\quad
9\coloneqq 8+1,\quad
10\coloneqq 9+1
\end{gathered}}

\section{Ziffernfolgen und Stellenwert}

\FormulaDefAuto[Dezimale Ziffernfolge in B04]{k\in\mathbb{N}\vdash \DecDigitSeq{d}{k}\coloneqq \bigl(d_0,\ldots,d_k\in\DecDigitSet\bigr)}

\FormulaDefAuto[Rekursive dezimale Stellenwertauswertung]{\DecDigitSeq{d}{k}\vdash \bigl(\DecPlaceVal{d}{0}=d_0\land \DecPlaceVal{d}{k+1}=\DecPlaceVal{d}{k}+d_{k+1}\cdot 10^{k+1}\bigr)}

\begin{theorem}[Summenform der Stellenwertauswertung]
\[
  \DecDigitSeq{d}{k}\vdash
  \DecPlaceVal{d}{k}=\sum_{i=0}^{k} d_i\cdot 10^i.
\]
\begin{tabproof}
  \proofcase[Basis]{k=0}
  \proofstep{}{\DecPlaceVal{d}{0}=d_0}{\FormulaRefAuto{\DecDigitSeq{d}{k}\vdash \bigl(\DecPlaceVal{d}{0}=d_0\land \DecPlaceVal{d}{k+1}=\DecPlaceVal{d}{k}+d_{k+1}\cdot 10^{k+1}\bigr)}}
  \proofcase[Schritt]{k\mapsto k+1}
  \proofstep{2}{\DecPlaceVal{d}{k}=\sum_{i=0}^{k}d_i\cdot10^i}{\rA}
  \proofstep{2}{\DecPlaceVal{d}{k+1}=\DecPlaceVal{d}{k}+d_{k+1}\cdot10^{k+1}}{\FormulaRefAuto{\DecDigitSeq{d}{k}\vdash \bigl(\DecPlaceVal{d}{0}=d_0\land \DecPlaceVal{d}{k+1}=\DecPlaceVal{d}{k}+d_{k+1}\cdot 10^{k+1}\bigr)}}
  \proofstep{2}{\DecPlaceVal{d}{k+1}=\sum_{i=0}^{k+1}d_i\cdot10^i}{\text{Definition der endlichen Summe}}

\FormulaDefAuto[Dezimales Zahlzeichen in B04]{\DecDigitSeq{d}{k}\vdash \DecNum{d}{k}\coloneqq\DecPlaceVal{d}{k}}

Ein sichtbares dezimales Zahlzeichen
\[
  d_kd_{k-1}\cdots d_0
\]
bezeichnet damit die natürliche Zahl
\[
  d_k\cdot10^k+d_{k-1}\cdot10^{k-1}+\dots+d_0.
\]
Kurze Beispiele sind
\[
  0=\DecPlaceVal{d}{0}\text{ bei }d_0=0,\qquad
  1=\DecPlaceVal{d}{0}\text{ bei }d_0=1,
\]
\[
  10=1\cdot10+0,\qquad
  123=1\cdot10^2+2\cdot10+3.
\]
