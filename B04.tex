\chapter{Einleitung}

Band~03 konstruiert die natürlichen Zahlen innerhalb der Mengenlehre:
\(0\) wird dort durch die leere Menge realisiert, und der Nachfolger entsteht
als \(x\cup\{x\}\). Wir übernehmen in diesem Band sofort das Zahlzeichen
\(0\) und verwenden die leere Menge danach nicht mehr als Zahlzeichen.

\FormulaDefAuto[Nullzeichen in B04]{0\coloneqq\varnothing}

Dieser Band wählt einen Peano-Ausgangspunkt. Wir betrachten eine Struktur mit
den Zeichen
\[
  \mathbb{N},\qquad 0,\qquad \Succ(n).
\]
Das Zeichen \(\Succ(n)\) ist das Nachfolgerzeichen; es ist noch keine Addition
mit der Zahl \(1\). Die Beweise der Peano-Axiome aus der mengentheoretischen
Konstruktion bleiben in Band~03. In Band~04 werden diese Eigenschaften als
Axiome gesammelt und anschließend einzeln referenzierbar gemacht.

\chapter{Peano-Axiome}

Die folgenden Axiome legen die Struktur \((\mathbb{N},0,n\mapsto \Succ(n))\)
fest. Das Nachfolgerzeichen wird im ganzen Abschnitt als primitives
einstelliges Operationszeichen gelesen.

\FormulaDefDeltaK[Peano-Axiome mit Nachfolgerfunktion]{%
  (\mathbb{N},0,\Succ)%
}{Peano-Axiome}{%
  \DeltaRow{\textbf{Axiome}}{}
  \DeltaRow{Null}
           {0\in\mathbb{N}}
           [\FormulaRefAutoFwd{0\in\mathbb{N}}]
  \DeltaRow{Nachfolgerabschluss}
           {n\in\mathbb{N}\vdash \Succ(n)\in\mathbb{N}}
           [\FormulaRefAutoFwd{n\in\mathbb{N}\vdash \Succ(n)\in\mathbb{N}}]
  \DeltaRow{Kein Nachfolger ist Null}
           {n\in\mathbb{N}\vdash \Succ(n)\neq 0}
           [\FormulaRefAutoFwd{n\in\mathbb{N}\vdash \Succ(n)\neq 0}]
  \DeltaRow{Injektivität}
           {n\in\mathbb{N}\dsep m\in\mathbb{N}\dsep \Succ(n)=\Succ(m)\vdash n=m}
           [\FormulaRefAutoFwd{n\in\mathbb{N}\dsep m\in\mathbb{N}\dsep \Succ(n)=\Succ(m)\vdash n=m}]
  \DeltaRow{Induktion}
           {P(0)\dsep \forall n\in\mathbb{N}\bigl(P(n)\rightarrow P(\Succ(n))\bigr)\vdash \forall n\in\mathbb{N}\,P(n)}
           [\FormulaRefAutoFwd{P(0)\dsep \forall n\in\mathbb{N}\bigl(P(n)\rightarrow P(\Succ(n))\bigr)\vdash \forall n\in\mathbb{N}\,P(n)}]
  \DeltaRow{\textbf{Neue Symbole}}{}
  \DeltaRow{Natürliche Zahlen}{\mathbb{N}}
  \DeltaRow{Nullzeichen}{0}[\FormulaRefAuto{0\coloneqq\varnothing}]
  \DeltaRow{Nachfolgerfunktion}{\Succ}
}

\FormulaAxiomAuto[Null ist natürlich]{0\in\mathbb{N}}

\FormulaAxiomAuto[Nachfolgerabschluss]{n\in\mathbb{N}\vdash \Succ(n)\in\mathbb{N}}

\FormulaAxiomAuto[Kein Nachfolger ist Null]{n\in\mathbb{N}\vdash \Succ(n)\neq 0}

\FormulaAxiomAuto[Injektivität des Nachfolgers]{n\in\mathbb{N}\dsep m\in\mathbb{N}\dsep \Succ(n)=\Succ(m)\vdash n=m}

\FormulaAxiomAuto[Induktionsaxiom]{P(0)\dsep \forall n\in\mathbb{N}\bigl(P(n)\rightarrow P(\Succ(n))\bigr)\vdash \forall n\in\mathbb{N}\,P(n)}

Die Existenz und Eindeutigkeit eines Nachfolgers ist in dieser Schreibweise
bereits durch die Termbildung \(\Succ(n)\) und den Nachfolgerabschluss
enthalten. Wir formulieren sie deshalb nicht zusätzlich als eigenes Axiom.

\chapter{Rekursion}

\begin{definition}[Rekursiv geschlossene Relationen]
Seien \(A\) eine Menge, \(a_0\in A\) und \(s\colon A\to A\). Eine Relation
\(R\subseteq\mathbb{N}\times A\) heißt zu \((a_0,s)\) rekursiv geschlossen,
wenn
\[
  (0,a_0)\in R
  \quad\text{und}\quad
  (n,a)\in R\vdash (\Succ(n),s(a))\in R
\]
für alle \(n\in\mathbb{N}\) und \(a\in A\) gelten.
\end{definition}

\FormulaDefAuto[Rekursionskern]{\mathcal{R}_{a_0,s}\coloneqq \bigcap\{R\subseteq\mathbb{N}\times A\mid R\text{ ist zu }(a_0,s)\text{ rekursiv geschlossen}\}}

\FormulaThmAuto[Dedekindscher Rekursionssatz]{a_0\in A\dsep s\colon A\to A\vdash \exists! f\colon\mathbb{N}\to A\,\Bigl(f(0)=a_0\land \forall n\in\mathbb{N}\bigl(f(\Succ(n))=s(f(n))\bigr)\Bigr)}
\begin{tabproof}
  \proofstep{1}{a_0\in A}{\rA}
  \proofstep{2}{s\colon A\to A}{\rA}
  \proofstep{1,2}{\mathcal{R}_{a_0,s}\subseteq\mathbb{N}\times A}{\FormulaRefAuto{\mathcal{R}_{a_0,s}\coloneqq \bigcap\{R\subseteq\mathbb{N}\times A\mid R\text{ ist zu }(a_0,s)\text{ rekursiv geschlossen}\}}}
  \proofstep{1,2}{(0,a_0)\in\mathcal{R}_{a_0,s}}{\FormulaRefAuto{\mathcal{R}_{a_0,s}\coloneqq \bigcap\{R\subseteq\mathbb{N}\times A\mid R\text{ ist zu }(a_0,s)\text{ rekursiv geschlossen}\}}}
  \proofstep{1,2}{(n,a)\in\mathcal{R}_{a_0,s}\vdash (\Succ(n),s(a))\in\mathcal{R}_{a_0,s}}{\FormulaRefAuto{\mathcal{R}_{a_0,s}\coloneqq \bigcap\{R\subseteq\mathbb{N}\times A\mid R\text{ ist zu }(a_0,s)\text{ rekursiv geschlossen}\}}}
