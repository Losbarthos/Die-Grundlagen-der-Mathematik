% ------------------------------------------------------------
% Kardinalität / Gleichmächtigkeit / Endlichkeit
% ------------------------------------------------------------
\providecommand{\EqCard}{\mathrel{\approx}}

\makeatletter
\def\Finite{%
  \@ifnextchar({\Finite@args}{\Finite@name}%
}
\def\Finite@name{%
  \ensuremath{\mathsf{Finite}}%
}
\def\Finite@args(#1){%
  \ensuremath{\mathsf{Finite}\!\left(#1\right)}%
}
\makeatother

\DeclareRobustCommand{\Infinite}[1]{%
  \ensuremath{\mathsf{Infinite}\!\left(#1\right)}%
}

\makeatletter
\def\Card{%
  \@ifnextchar({\Card@args}{\Card@name}%
}
\def\Card@name{%
  \ensuremath{\mathsf{Card}}%
}
\def\Card@args(#1){%
  \ensuremath{\mathsf{Card}\!\left(#1\right)}%
}

\def\tCard{%
  \@ifnextchar({\tCard@args}{\tCard@name}%
}
\def\tCard@name{%
  \ensuremath{t_{\mathsf{Card}}}%
}
\def\tCard@args(#1){%
  \ensuremath{t_{\mathsf{Card}}\!\left(#1\right)}%
}
\makeatother
