\ProvidesFile{cardinality.tex}[2026/03/25 Cardinality and finiteness notation]

\providecommand{\EqCard}{\mathrel{\approx}}

\DeclareRobustCommand{\Finite}[1]{%
  \ensuremath{\mathsf{Finite}\!\left(#1\right)}%
}

\DeclareRobustCommand{\Infinite}[1]{%
  \ensuremath{\mathsf{Infinite}\!\left(#1\right)}%
}

\DeclareRobustCommand{\Card}[1]{%
  \ensuremath{\mathsf{Card}\!\left(#1\right)}%
}

\makeatletter
\DeclareRobustCommand{\tCard}{%
  \@ifnextchar({\tCard@paren}{%
    \@ifnextchar\bgroup{\tCard@brace}{\ensuremath{t_{\mathsf{Card}}}}%
  }%
}
\def\tCard@paren(#1){%
  \ensuremath{t_{\mathsf{Card}}\!\left(#1\right)}%
}
\def\tCard@brace#1{%
  \ensuremath{t_{\mathsf{Card}}\!\left(#1\right)}%
}
\makeatother

\DeclareRobustCommand{\CardValues}[1]{%
  \ensuremath{\mathsf{CardVals}\!\left(#1\right)}%
}
