% tex/impl/commands/functions.tex
\ProvidesFile{functions.tex}[2026/01/04 Function-theoretic notation]

% ------------------------------------------------------------
% Pfeile für Abbildungen
% ------------------------------------------------------------
\providecommand{\inj}{\rightarrowtail}           % injektiv
\providecommand{\sur}{\twoheadrightarrow}        % surjektiv
\providecommand{\bij}{\xrightarrow{\sim}}        % bijektiv

% ------------------------------------------------------------
% Standard-Namen / Operatoren
% ------------------------------------------------------------
\providecommand{\Id}{\mathrm{id}}                % Identitäts-Operator (z.B. \Id_A)
\providecommand{\Graph}{\ensuremath{\mathrm{Graph}}}

% Induzierte Potenzmengenabbildung
\DeclareRobustCommand{\PowMap}[1]{\mathcal{P}_{#1}}
\DeclareRobustCommand{\tPowMap}[1]{t_{\PowMap{#1}}}

\DeclareMathOperator{\Fib}{Fib}                  % Faser (Operator)
\DeclareMathOperator{\Sec}{Sec}                  % Sektion / Rechtsinverse (z.B. \Sec_{G,L})
\DeclareRobustCommand{\ChoicePow}[2]{%
  \ensuremath{\mathsf{ChoicePow}\!\left(#1,#2\right)}%
}
\DeclareRobustCommand{\ACsur}{\ensuremath{\mathsf{AC}_{\mathrm{sur}}}}

% ------------------------------------------------------------
% Graph einer Surjektion aus einer Injektion (mit Stützelement a_0)
% ------------------------------------------------------------
\DeclareRobustCommand{\Gsurjfrominj}[2]{G_{#1,#2}}

% commands/functions.tex
\DeclareMathOperator{\Succ}{succ} 
\DeclareMathOperator{\Pred}{pred}

\DeclareRobustCommand{\Jomit}[1]{J_{#1}}
\DeclareRobustCommand{\tJomit}[1]{t_{J_{#1}}}

\DeclareMathOperator{\Ext}{Ext}

\newcommand{\tExtMap}[3]{t_{\Ext_{#1,#2,#3}}}
\newcommand{\ExtMap}[3]{\Ext_{#1,#2,#3}}

% Potenzmengenadjunktion natürlicher Zahlen
\DeclareRobustCommand{\PowAdj}[1]{S^{\mathcal P}_{#1}}
\DeclareRobustCommand{\tPowAdj}[1]{t_{\PowAdj{#1}}}

\DeclareRobustCommand{\ZeroFix}[1]{Z_{#1}}

\DeclareRobustCommand{\RecFun}[2]{\mathrm{Rec}_{#1,#2}}

% Konstante Funktionen.
\DeclareRobustCommand{\ConstFun}[3]{K^{#1,#2}_{#3}}
\DeclareRobustCommand{\tConstFun}[3]{t_{\ConstFun{#1}{#2}{#3}}}

% Fallabbildungen.
\DeclareRobustCommand{\CaseFun}[4]{%
  \left\{\begin{array}{@{}ll@{}}
    #3, & \text{auf } #1\\
    #4, & \text{auf } #2
  \end{array}\right.%
}
\DeclareRobustCommand{\tCaseFun}[4]{t_{\CaseFun{#1}{#2}{#3}{#4}}}

% Additiver Schritt fuer die Multiplikationsrekursion: sigma_a(x)=x+a.
\DeclareRobustCommand{\AddStep}[1]{\sigma_{#1}}
\DeclareRobustCommand{\tAddStep}[1]{t_{\AddStep{#1}}}

% Zustandsuebergaenge fuer endliche Summen und Produkte.
\DeclareRobustCommand{\SumStep}[1]{\sigma^\Sigma_{#1}}
\DeclareRobustCommand{\tSumStep}[1]{t_{\SumStep{#1}}}
\DeclareRobustCommand{\ProdStep}[1]{\sigma^\Pi_{#1}}
\DeclareRobustCommand{\tProdStep}[1]{t_{\ProdStep{#1}}}

% Nachfolgerverschiebung entlang einer eingebetteten natürlichen Folge
\DeclareRobustCommand{\SuccShift}[1]{S^{+}_{#1}}
\DeclareRobustCommand{\tSuccShift}[1]{t_{\SuccShift{#1}}}

% Natuerliche und korrigierte Adjunktionsabbildungen in Mengenfamilien.
\DeclareRobustCommand{\FranklAdj}[2]{J^{#1}_{#2}}
\DeclareRobustCommand{\tFranklAdj}[2]{t_{\FranklAdj{#1}{#2}}}
