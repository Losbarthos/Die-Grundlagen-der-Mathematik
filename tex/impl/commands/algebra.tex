\ProvidesFile{algebra.tex}[2026/07/18 Algebraic and semilattice notation]

% Algebraische Halbverbands- und Verbandsprädikate.
\DeclareRobustCommand{\BinOp}[1]{%
  \ensuremath{\mathbin{#1}}%
}
\DeclareRobustCommand{\JoinSemi}[1]{%
  \ensuremath{\mathsf{JoinSemi}\!\left(#1\right)}%
}
\DeclareRobustCommand{\MeetSemi}[1]{%
  \ensuremath{\mathsf{MeetSemi}\!\left(#1\right)}%
}
\DeclareRobustCommand{\LatticeAlg}[1]{%
  \ensuremath{\mathsf{Lattice}\!\left(#1\right)}%
}

% Von einer Halbverbandsoperation induzierte Ordnungsrelationen.
\DeclareRobustCommand{\JoinOrd}[1]{%
  \ensuremath{R^{\sqcup}_{#1}}%
}
\DeclareRobustCommand{\MeetOrd}[1]{%
  \ensuremath{R^{\sqcap}_{#1}}%
}

% Universelle Eigenschaften zweier Elemente.
\DeclareRobustCommand{\PairSup}[5]{%
  \ensuremath{\mathsf{PairSup}_{#1,#2}\!\left(#3,#4;#5\right)}%
}
\DeclareRobustCommand{\PairInf}[5]{%
  \ensuremath{\mathsf{PairInf}_{#1,#2}\!\left(#3,#4;#5\right)}%
}
\DeclareRobustCommand{\PairJoinOp}[1]{%
  \ensuremath{\mathsf{PairJoinOp}\!\left(#1\right)}%
}
\DeclareRobustCommand{\PairMeetOp}[1]{%
  \ensuremath{\mathsf{PairMeetOp}\!\left(#1\right)}%
}

% Null-Halbverbände und Irreduzibilität.
\DeclareRobustCommand{\ZeroJoinSemi}[1]{%
  \ensuremath{\mathsf{ZeroJoinSemi}\!\left(#1\right)}%
}
\DeclareRobustCommand{\JoinIrr}[2]{%
  \ensuremath{\mathsf{JoinIrr}_{#1}\!\left(#2\right)}%
}
\DeclareRobustCommand{\JoinIrrSet}[1]{%
  \ensuremath{\mathsf{Jirr}\!\left(#1\right)}%
}

% Hauptfilter und Hauptideal in einer dargestellten Ordnung.
\DeclareRobustCommand{\PrincipalFilter}[2]{%
  \ensuremath{\mathord{\uparrow}_{#1}\!#2}%
}
\DeclareRobustCommand{\PrincipalIdeal}[2]{%
  \ensuremath{\mathord{\downarrow}_{#1}\!#2}%
}

% Injektive Form von "höchstens die Hälfte" und Frankls Halbverbandsform.
\DeclareRobustCommand{\AtMostHalf}[2]{%
  \ensuremath{\mathsf{AtMostHalf}\!\left(#1;#2\right)}%
}
\DeclareRobustCommand{\RareOcc}[2]{%
  \ensuremath{\mathsf{RareOcc}\!\left(#1,#2\right)}%
}
\DeclareRobustCommand{\FranklSemi}[1]{%
  \ensuremath{\mathsf{FranklSemi}\!\left(#1\right)}%
}
\DeclareRobustCommand{\InterClosedFam}[1]{%
  \ensuremath{\mathsf{InterClosedFam}\!\left(#1\right)}%
}

% Kanonische Familie der unter einem Element liegenden Join-Irreduziblen.
\DeclareRobustCommand{\IrrBelow}[2]{%
  \ensuremath{\mathsf{JirrBelow}_{#1}\!\left(#2\right)}%
}
\DeclareRobustCommand{\CanonicalMeetFamily}[1]{%
  \ensuremath{\mathsf{CanMeetFam}\!\left(#1\right)}%
}

% Namen der drei globalen Formen von Frankls Vermutung.
\DeclareRobustCommand{\FranklSetStatement}{\ensuremath{\mathsf{FC}_{\mathrm{set}}}}
\DeclareRobustCommand{\FranklInterStatement}{\ensuremath{\mathsf{FC}_{\cap}}}
\DeclareRobustCommand{\FranklJoinStatement}{\ensuremath{\mathsf{FC}_{\sqcup}}}
